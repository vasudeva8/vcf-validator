33 size_t current_line = 0; // the first line is the number 1, ParsingState takes this convention too
34
35 size_t errors = errorDAO.count_errors();
36 size_t errors_fixed = 0;
37 if (errors == 0) {
38 BOOST_LOG_TRIVIAL(info) << "The errors report was empty, there are no errors to fix the input";
39 return 0;
40 }
41
42 ebi::vcf::Fixer fixer{output};
43
44 errorDAO.for_each_error([&](std::shared_ptr<ebi::vcf::Error> error) {
45 size_t line_index = error->line;
46 while (current_line < line_index) {
47
48 // advance input
49 if (ebi::util::readline(input, line).size() == 0) {
50 throw std::runtime_error("The file was shorter than expected, only "
51 + std::to_string(errors_fixed) + "/" + std::to_string(errors)
52 + " error reports were processed");
53 }
54 current_line++;
55 if (current_line == line_index) {
56 break;
57 }
58 ebi::util::writeline(output, line);
59 }
60 fixer.fix(line_index, line, *error);
61 ++errors_fixed;
62 });
63
64 // advance input from the last error to the end of input
65 while (ebi::util::readline(input, line).size() != 0) {
66 for (auto c : line) {
67 output << c;
68 }
69 }
70
71 size_t ignored_errors = fixer.get_ignored_errors();
72 if (ignored_errors != 0) {
73 BOOST_LOG_TRIVIAL(info) << "There were " << ignored_errors << " errors that couldn't be automatically fixed";
74 }
75
76 return ignored_errors;
77 }

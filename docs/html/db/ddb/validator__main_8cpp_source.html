97 std::vector<std::unique_ptr<ebi::vcf::ReportWriter>> get_outputs(std::string const &output_str, std::string const &input)
98 {
99 std::vector<std::string> outs;
100 ebi::util::string_split(output_str, ",", outs);
101 size_t initial_size = outs.size();
102
103 // don't write several times to the same output
104 std::sort(outs.begin(), outs.end());
105 std::unique(outs.begin(), outs.end());
106 if (initial_size != outs.size()) {
107 BOOST_LOG_TRIVIAL(warning) << "Duplicated outputs! will write just once to each output specified by -r/--report";
108 }
109
110 std::vector<std::unique_ptr<ebi::vcf::ReportWriter>> outputs;
111
112 auto epoch = std::chrono::system_clock::now().time_since_epoch();
113 auto timestamp = std::chrono::duration_cast<std::chrono::milliseconds>(epoch).count();
114 for (auto out : outs) {
115 if (out == ebi::vcf::DATABASE || out == ebi::vcf::TEXT || out == ebi::vcf::SUMMARY) {
116 std::string filetype = (out == ebi::vcf::DATABASE ? "db" : "txt");
117 std::string errortype = (out == ebi::vcf::SUMMARY) ? "errors_summary" : "errors";
118 std::string filename = input + "." + errortype + "." + std::to_string(timestamp) + "." + filetype;
119 boost::filesystem::path file{filename};
120 if (boost::filesystem::exists(file)) {
121 throw std::runtime_error{"Report file already exists on " + filename + ", please delete it or rename it"};
122 }
123 if (out == ebi::vcf::DATABASE) {
124 outputs.emplace_back(new ebi::vcf::OdbReportRW(filename));
125 } else if (out == ebi::vcf::TEXT) {
126 outputs.emplace_back(new ebi::vcf::FileReportWriter(filename));
127 } else {
128 outputs.emplace_back(new ebi::vcf::SummaryReportWriter(filename));
129 }
130 } else {
131 throw std::invalid_argument{"Please use only valid report types"};
132 }
133 }
134
135 return outputs;
136 }

170 }

172

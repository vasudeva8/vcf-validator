39 template<typename C>
40 void string_split(std::string const & s, char const * delims, C & ret)
41 {
42 C output;
43
44 if (s.size() > 0) {
45 char const* p = s.c_str();
46 char const* q = strpbrk(p+1, delims);
47
48 // Insert first to last-1 elements
49 for( ; q != NULL; q = strpbrk(p, delims) )
50 {
51 output.push_back(typename C::value_type(p, q));
52 p = q + 1;
53 }
54
55 // Insert last element
56 if (p < &(s.back()) + 1) {
57 output.push_back(typename C::value_type(p));
58 }
59 }
60
61 output.swap(ret);
62 }

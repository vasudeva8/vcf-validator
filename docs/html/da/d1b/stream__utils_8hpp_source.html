124 template <typename T>
125 std::ostream &print_container(std::ostream &os, const T &container,
126 std::string open_tag, std::string separator_tag, std::string close_tag) {
127 os << open_tag;
128 auto it = container.begin();
129 auto end = container.end();
130 if (it != end) {
131 os << *it;
132 for (++it; it != container.end(); ++it) {
133 os << separator_tag << *it;
134 }
135 }
136 os << close_tag;
137 return os;
138 }

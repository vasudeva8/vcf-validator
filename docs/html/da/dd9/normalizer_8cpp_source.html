26 bool RecordCore::operator<(const RecordCore &other) const
27 {
28 int compare = chromosome.compare(other.chromosome); // <0: a<b; 0: a==b; >0: a>b
29 if (compare != 0) {
30 return compare < 0;
31 }
32 // else: chromosome is the same
33
34 if (position != other.position) {
35 return position < other.position;
36 }
37 compare = reference_allele.compare(other.reference_allele);
38 if (compare != 0) {
39 return compare < 0;
40 }
41 compare = alternate_allele.compare(other.alternate_allele);
42 return compare < 0;
43 }

77 std::vector<RecordCore> normalize(const Record &record/* , ParsingState?*/)
78 {
79 std::vector<RecordCore> records;
80
81 // This index is necessary for getting the samples where the mutated allele is present
82 for (size_t i = 0; i < record.alternate_alleles.size(); i++) {
83 std::string alternate = record.alternate_alleles[i];
84 std::string reference = record.reference_allele;
85 size_t position = record.position;
86
87 size_t corrected_position;
88 std::string corrected_alternate;
89 std::string corrected_reference;
90
91 // assertions / preconditions
92 if (alternate.size() < 1) {
93 throw new NormalizationError{record.line, "Alternate should not be empty"};
94 }
95 if (reference.size() < 1) {
96 throw new NormalizationError{record.line, "Reference should not be empty"};
97 }
98 if (reference == alternate) {
99 throw new NormalizationError{record.line, "Reference and alternate should not be identical"};
100 }
101
102 // count trailing matching bases using mismatch with reverse iterators
103 // the result are reverse_iterators, which later must be made forward iterators
104 std::pair<std::string::reverse_iterator, std::string::reverse_iterator> trail_mismatch_reverse =
105 util::mismatch(reference.rbegin(), reference.rend(),
106 alternate.rbegin(), alternate.rend());
107 // trail_match_* point to the first match of the trailing bases. These will be the "end" in
108 // the .assign [start, end) below. "advance" because trail_mismatch_* were pointing to the last MISmatch
109 std::string::iterator trail_match_ref = make_forward_and_advance(trail_mismatch_reverse.first);
110 std::string::iterator trail_match_alt = make_forward_and_advance(trail_mismatch_reverse.second);
111
112
113 // these will point to the first leading mismatch. These will be the "start" in the .assign [start, end)
114 std::pair<std::string::iterator, std::string::iterator> lead_mismatch_indices =
115 util::mismatch(reference.begin(), trail_match_ref,
116 alternate.begin(), trail_match_alt);
117
118
119 // using the previous iterators, assign only [first_mismatch, last_mismatch+1)
120 corrected_reference.assign(lead_mismatch_indices.first, trail_match_ref);
121 corrected_alternate.assign(lead_mismatch_indices.second, trail_match_alt);
122 corrected_position = position + (lead_mismatch_indices.first - reference.begin());
123
124 records.emplace_back(record.line, record.chromosome, corrected_position,
125 corrected_reference, corrected_alternate, record.types[i]);
126 }
127
128 return records;
129 }

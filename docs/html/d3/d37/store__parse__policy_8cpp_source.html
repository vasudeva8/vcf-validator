54 void StoreParsePolicy::handle_fileformat(ParsingState & state)
55 {
56 Version fileformat_version = Version::v41;
57
58 if (m_current_token == VCF_V41) {
59 fileformat_version = Version::v41;
60 } else if (m_current_token == VCF_V42) {
61 fileformat_version = Version::v42;
62 } else if (m_current_token == VCF_V43) {
63 fileformat_version = Version::v43;
64 } else if (m_current_token == VCF_V44) {
65 fileformat_version = Version::v44;
66 } else {
67 throw new FileformatError{state.n_lines, "Not allowed VCF fileformat version"};
68 }
69
70 if (fileformat_version != state.source->version) {
71 throw new FileformatError{state.n_lines, "Unexpected VCF fileformat version found"};
72 } else {
73 state.set_version(fileformat_version);
74 }

88 void StoreParsePolicy::handle_meta_line(ParsingState & state)
89 {
90 // Put together m_line_typeid and m_grouped_tokens in a single MetaEntry object
91 // Add MetaEntry to Source
92
93 if (m_line_typeid == "") { // Plain value
94 state.add_meta(MetaEntry{state.n_lines, m_grouped_tokens[0], state.source});
95
96 } else if (m_grouped_tokens.size() == 1) { // TypeID=value
97 state.add_meta(MetaEntry{state.n_lines, m_line_typeid, m_grouped_tokens[0], state.source});
98
99 } else if (m_grouped_tokens.size() % 2 == 0) { // TypeID=<Key-value pairs>
100 auto key_values = std::map<std::string, std::string>{};
101 for (size_t i = 0; i < m_grouped_tokens.size(); i += 2) {
102 key_values[m_grouped_tokens[i]] = m_grouped_tokens[i+1];
103 }
104 state.add_meta(MetaEntry{state.n_lines, m_line_typeid, key_values, state.source});
105
106 } else {
107 throw new MetaSectionError{state.n_lines, "Meta line description is not a value, nor a TypeID=value, nor a TypeID=<Key-value pairs>"};
108 }
109 }

123 void StoreParsePolicy::check_each_sample_appears_only_once(ParsingState const &state) const
124 {
125 using util::operator<<;
126 std::set<std::string> unique_samples;
127 std::map<std::string, int> repeated_sample_names;
128 for (const auto &sample_name : m_grouped_tokens) {
129 auto inserted = unique_samples.insert(sample_name);
130 bool is_inserted = inserted.second;
131
132 if (!is_inserted) {
133 repeated_sample_names[sample_name]++;
134 }
135 }
136 if (!repeated_sample_names.empty()) {
137 throw new HeaderSectionError{state.n_lines, build_duplicated_samples_message(repeated_sample_names,
138 m_grouped_tokens.size() - unique_samples.size())};
139 }
140 }

164 complete_message.pop_back();

168 void StoreParsePolicy::handle_column_end(ParsingState const & state, size_t n_columns)
169 {
170 switch(n_columns) {
171 case 1:
172 m_line_tokens[CHROM] = m_grouped_tokens;
173 break;
174 case 2:
175 m_line_tokens[POS] = m_grouped_tokens;
176 break;
177 case 3:
178 m_line_tokens[ID] = m_grouped_tokens;
179 break;
180 case 4:
181 m_line_tokens[REF] = m_grouped_tokens;
182 break;
183 case 5:
184 m_line_tokens[ALT] = m_grouped_tokens;
185 break;
186 case 6:
187 m_line_tokens[QUAL] = m_grouped_tokens;
188 break;
189 case 7:
190 m_line_tokens[FILTER] = m_grouped_tokens;
191 break;
192 case 8:
193 m_line_tokens[INFO] = m_grouped_tokens;
194 break;
195 case 9:
196 m_line_tokens[FORMAT] = m_grouped_tokens;
197 break;
198 default:
199 // Collection of samples
200 if (m_line_tokens.find(SAMPLES) == m_line_tokens.end()) {
201 m_line_tokens[SAMPLES] = std::vector<std::string>{};
202 }
203 // Samples are stored as a single string
204 m_line_tokens[SAMPLES].push_back(m_grouped_tokens[0]);
205 }
206 m_grouped_tokens = std::vector<std::string>{};

209 void StoreParsePolicy::handle_body_line(ParsingState & state)
210 {
211 size_t position;
212 try {
213 // Transform the position token into a size_t
214 position = static_cast<size_t>(std::stoi(m_line_tokens[POS][0]));
215 } catch (std::invalid_argument ex) {
216 throw new PositionBodyError{state.n_lines};
217 }
218
219 // Transform all the quality tokens into floating point numbers
220 float quality = 0;
221 if (m_line_tokens[QUAL][0] != MISSING_VALUE) {
222 try {
223 quality = std::stof(m_line_tokens[QUAL][0]);
224 } catch (std::invalid_argument ex) {
225 throw new QualityBodyError{state.n_lines};
226 }
227 }
228
229 // Split the info tokens by the equals (=) symbol
230 std::multimap<std::string, std::string> info;
231 for (auto &field : m_line_tokens[INFO]) {
232 std::vector<std::string> subfields;
233 util::string_split(field, "=", subfields);
234 if (subfields.size() > 1) {
235 info.emplace(subfields[0], subfields[1]);
236 } else {
237 info.emplace(subfields[0], "");
238 }
239 }
240
241 // Format and samples are optional
242 auto format = m_line_tokens.find(FORMAT) != m_line_tokens.end() ?
243 m_line_tokens[FORMAT] : std::vector<std::string>{};
244 auto samples = m_line_tokens.find(SAMPLES) != m_line_tokens.end() ?
245 m_line_tokens[SAMPLES] : std::vector<std::string>{};
246
247 state.set_record(std::unique_ptr<Record>{new Record{
248 state.n_lines,
249 m_line_tokens[CHROM][0],
250 position,
251 m_line_tokens[ID],
252 m_line_tokens[REF][0],
253 m_line_tokens[ALT],
254 quality,
255 m_line_tokens[FILTER],
256 info,
257 format,
258 samples,
259 state.source
260 }});
261
262 check_sorted(state, position);
263 }

278 void StoreParsePolicy::check_sorted(ParsingState &state, size_t position)
279 {
280 // check contigs are contiguous
281 auto iterator = finished_contigs.find(m_line_tokens[CHROM][0]);
282 bool contig_not_found = iterator == finished_contigs.end();
283 bool contig_already_finished = contig_not_found ? false : iterator->second;
284 if (contig_not_found) {
285 // contig not found in the map: finishing the previous contig, and starting a new one
286 if (finished_contigs.size() != 0) {
287 // with the first contig there's no previous contig
288 finished_contigs[previous_contig] = true;
289 }
290 finished_contigs[m_line_tokens[CHROM][0]] = false;
291 previous_contig = m_line_tokens[CHROM][0];
292 previous_position = 0; // position sorting is reset
293 } else if (contig_already_finished) {
294 std::stringstream ss, ss_detail;
295 ss << "Variant is not contiguous to the rest of the contig";
296 ss_detail << "Position of variant " << m_line_tokens[CHROM][0] << " : " << position;
297 throw new BodySectionError{state.n_lines, ss.str(), ss_detail.str()};
298 }
299
300 // check all positions are sorted within a contig
301 if (position < previous_position) {
302 std::stringstream ss, ss_detail;
303 ss << "Contig is not sorted by position";
304 ss_detail << "Contig " << m_line_tokens[CHROM][0] << " position " << position << " found after " << previous_position;
305 throw new PositionBodyError{state.n_lines, ss.str(), ss_detail.str()};
306 }
307 previous_position = position;
308 }

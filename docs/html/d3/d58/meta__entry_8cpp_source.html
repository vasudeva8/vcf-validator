90 void MetaEntryVisitor::operator()(std::map<std::string, std::string> & value) const
91 {
92 auto & id = entry.id;
93 if (id == ALT) {
94 check_alt(value);
95 } else if (id == ASSEMBLY) {
96 // TODO May check URL correctness (regexp?)
97 } else if (id == CONTIG) {
98 check_contig(value);

103 } else if (id == INFO) {
104 check_info(value);
105 } else if (id == PEDIGREE) {

130 void MetaEntryVisitor::check_alt_id(std::string const & id_field) const
131 {
132 // Check ID main type (prefix before ':') is one of the predefined ones
133 auto main_type_position_end = id_field.find(':');
134 bool colon_present = main_type_position_end != std::string::npos;
135 if (colon_present) {
136 auto main_type = id_field.substr(0, main_type_position_end);
137 if (!ebi::util::contains(PREDEFINED_INFO_SVTYPES, main_type)) {
138 std::stringstream message;
139 message << "In ALT metadata IDs containing colon-separated type and subtypes, the top level type "
140 "must be one of: ";
141 ebi::util::print_container(message, PREDEFINED_INFO_SVTYPES, "", ", ", "");
142 throw new MetaSectionError{entry.line, message.str(), "Found ID was '" + id_field + "'"};
143 }
144 }

169 void MetaEntryVisitor::check_format(std::map<std::string, std::string> & value) const
170 {
171 // It must contain an ID, Number, Type and Description
172 check_key_is_present(FORMAT, ID, value.count(ID));
173 check_key_is_present(FORMAT, NUMBER, value.count(NUMBER));
174 check_key_is_present(FORMAT, TYPE, value.count(TYPE));
175 check_key_is_present(FORMAT, DESCRIPTION, value.count(DESCRIPTION));
176
177 check_format_or_info_number(value[NUMBER], FORMAT);
178 check_format_type(value[TYPE]);
179
180 if (entry.source->version == Version::v41 || entry.source->version == Version::v42) {
181 check_predefined_tag(FORMAT, NUMBER, value, format_v41_v42);
182 check_predefined_tag(FORMAT, TYPE, value, format_v41_v42);
183 } else if (entry.source->version == Version::v43 ) {
184 check_predefined_tag(FORMAT, NUMBER, value, format_v43);
185 check_predefined_tag(FORMAT, TYPE, value, format_v43);
186 } else {
187 check_predefined_tag(FORMAT, NUMBER, value, format_v44);
188 check_predefined_tag(FORMAT, TYPE, value, format_v44);
189 }

247 void MetaEntryVisitor::check_predefined_tag(std::string const & tag_field, std::string const & meta_entry_property,
248 std::map<std::string, std::string> & meta_entry,
249 std::map<std::string, std::pair<std::string, std::string>> const & predefined_meta_entries) const
250 {
251 auto iterator = predefined_meta_entries.find(meta_entry[ID]);
252 if (iterator != predefined_meta_entries.end()) {
253 // Determine the required value of the key based on whether we are checking for Type or Number
254 std::string predefined_value = (meta_entry_property == TYPE ? get_predefined_type(iterator)
255 : get_predefined_number(iterator));
256 // If the required value is a "." (dot), do nothing
257 // Or if the required value does not match the value provided in the vcf file, throw an error
258 if (predefined_value != MISSING_VALUE && predefined_value != meta_entry[meta_entry_property]) {
259 std::string message = tag_field + " " + meta_entry[ID] + " metadata " + meta_entry_property + " is not " + predefined_value;
260 throw new MetaSectionError{entry.line, message, ErrorFix::RECOVERABLE_VALUE, meta_entry_property, predefined_value};
261 }
262 }

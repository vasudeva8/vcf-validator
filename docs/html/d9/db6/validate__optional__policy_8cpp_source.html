32 void ValidateOptionalPolicy::optional_check_body_entry(ParsingState & state, Record const & record) //const
33 {
34 // Position zero should only be used for telomeres
35 check_body_entry_position_zero(state, record);
36
37 // The standard separator is semi-colon, commas are accepted but most probably a mistake
38 check_body_entry_id_commas(state, record);
39
40 // Reference and alternate alleles in indels should share the first nucleotide
41 check_body_entry_reference_alternate_matching(state, record);
42
43 // 0/0 genotypes should be present when ALT is <*>, as it is supposed to be a reference region
44 check_body_entry_alt_gvcf_gt_value(state, record);
45
46 // gVCF fields should provide END, as <*> is supposed to represent a region
47 check_body_entry_info_gvcf_end(state, record);
48
49 // If a variant is flagged as precise, then it should not contain imprecise variant fields like CIPOS or CIEND
50 check_body_entry_info_imprecise(state, record);
51
52 // The number of values in SVLEN should match the number of alternate alleles
53 check_body_entry_info_svlen(state, record);
54
55 // Confidence interval tags should have first value <=0 and second value >= 0
56 check_body_entry_info_confidence_interval(state, record);
57
58 /*
59 * Once some meta-data is marked as in/correct there is no need again, so all the following have been
60 * optimised using a map for correctly defined meta-data and another one for incorrectly defined.
61 */
62
63 // The chromosome/contig should be described in the meta section
64 check_contig_meta(state, record);
65
66 // Alternate alleles of the form <SOME_ALT> should be described in the meta section
67 check_alternate_allele_meta(state, record);
68
69 // Filters should be described in the meta section
70 check_filter_meta(state, record);
71
72 // Info fields should be described in the meta section
73 check_info_meta(state, record);
74
75 // Format fields should be described in the meta section
76 check_format_meta(state, record);
77 }

101 void ValidateOptionalPolicy::check_body_entry_reference_alternate_matching(ParsingState & state, Record const & record)
102 {
103 for (size_t i = 0; i < record.alternate_alleles.size(); ++i) {
104 auto & alternate = record.alternate_alleles[i];
105 auto type = record.types[i];
106
107 if (type == RecordType::INDEL && alternate[0] != record.reference_allele[0]) {
108 throw new ReferenceAlleleBodyError{state.n_lines,
109 "Reference and alternate alleles do not share the first nucleotide"};
110 }

114 void ValidateOptionalPolicy::check_body_entry_alt_gvcf_gt_value(ParsingState & state, Record const & record) const
115 {
116 if (util::contains(record.alternate_alleles, GVCF_NON_VARIANT_ALLELE) && record.format[0] == vcf::GT) {
117 for (auto & sample : record.samples) {
118 if (sample_has_reference_in_all_alleles(sample)) {

128 bool ValidateOptionalPolicy::sample_has_reference_in_all_alleles(std::string const & sample) const
129 {
130 std::string gt_subfield = sample.substr(0, sample.find(':'));
131 std::vector<std::string> alleles;
132 util::string_split(gt_subfield, "/|", alleles);
133 for (auto & allele : alleles) {
134 if (allele != "0") {
135 return false;
136 }
137 }
138 return true;

170 void ValidateOptionalPolicy::check_body_entry_info_svlen(ParsingState & state, Record const & record) const
171 {
172 auto it = record.info.find(SVLEN);
173 if (it != record.info.end()) {
174 std::vector<std::string> values;
175 util::string_split(it->second, ",", values);
176 if (values.size() != record.alternate_alleles.size()) {
177 throw new InfoBodyError{state.n_lines,
178 "INFO SVLEN should have same number of values as ALT", "Expected " + std::to_string(record.alternate_alleles.size())
179 + ", found " + std::to_string(values.size())};

184 void ValidateOptionalPolicy::check_body_entry_info_confidence_interval(ParsingState & state, Record const & record) const
185 {
186 std::vector<std::string> confidence_interval_tags = { CICN, CICNADJ, CIEND, CILEN, CIPOS };
187 for (auto & confidence_interval_tag : confidence_interval_tags) {
188 auto it = record.info.find(confidence_interval_tag);
189 if (it != record.info.end()) {
190 std::vector<std::string> values;
191 util::string_split(it->second, ",", values);
192 size_t scanned_first_value_length, scanned_second_value_length;
193 int first_numeric_value = std::stoi(values[0], &scanned_first_value_length);
194 int second_numeric_value = std::stoi(values[1], &scanned_second_value_length);
195 if (first_numeric_value > 0 || second_numeric_value < 0
196 || values[0].size() != scanned_first_value_length || values[1].size() != scanned_second_value_length) {
197 throw new InfoBodyError{state.n_lines,
198 "INFO " + confidence_interval_tag +
199 " is a confidence interval tag, which should have first value <= 0 and second value >= 0"};
200 }
201 }

205 void ValidateOptionalPolicy::check_contig_meta(ParsingState & state, Record const & record) const
206 {
207 // The associated 'contig' meta entry should exist (notify only once)
208 std::string current_chromosome = record.chromosome;
209
210 if (state.is_well_defined_meta(CONTIG, current_chromosome)) {
211 return; // Check only once
212 }
213
214 std::pair<meta_iterator, meta_iterator> range = state.source->meta_entries.equal_range(CONTIG);
215
216 if (is_record_subfield_in_header(current_chromosome, range.first, range.second)) {
217 state.add_well_defined_meta(CONTIG, current_chromosome);
218 } else {
219 throw new NoMetaDefinitionError{
220 state.n_lines,
221 "Chromosome/contig '" + current_chromosome + "' is not described in a 'contig' meta description"
222 };
223 }

226 void ValidateOptionalPolicy::check_alternate_allele_meta(ParsingState & state, Record const & record) const
227 {
228 static boost::regex square_brackets_regex("<([a-zA-Z0-9:_]+)>");
229 std::pair<meta_iterator, meta_iterator> range = state.source->meta_entries.equal_range(ALT);
230 boost::cmatch pieces_match;
231
232 for (auto & alternate : record.alternate_alleles) {
233 // Check alternate ID is present in meta-entry (only applies to the form <SOME_ALT_ID>)
234 if (alternate[0] == '<' && boost::regex_match(alternate.c_str(), pieces_match, square_brackets_regex)) {
235 std::string alt_id = pieces_match[1];
236
237 if (state.is_well_defined_meta(ALT, alt_id)) {
238 continue; // Check only once
239 }
240
241 if (is_record_subfield_in_header(alt_id, range.first, range.second)) {
242 state.add_well_defined_meta(ALT, alt_id);
243 } else {
244 throw new NoMetaDefinitionError{
245 state.n_lines,
246 "Alternate '<" + alt_id + ">' is not listed in a valid meta-data ALT entry"
247 };
248 }
249 }
250 }

253 void ValidateOptionalPolicy::check_filter_meta(ParsingState & state, Record const & record) const
254 {
255 std::pair<meta_iterator, meta_iterator> range = state.source->meta_entries.equal_range(FILTER);
256
257 for (auto & filter : record.filters) {
258 if (filter == PASS || filter == MISSING_VALUE) { continue; } // No need to check PASS or missing data
259
260 if (state.is_well_defined_meta(FILTER, filter)) {
261 continue; // Check only once
262 }
263
264 if (is_record_subfield_in_header(filter, range.first, range.second)) {
265 state.add_well_defined_meta(FILTER, filter);
266 } else {
267 throw new NoMetaDefinitionError{
268 state.n_lines,

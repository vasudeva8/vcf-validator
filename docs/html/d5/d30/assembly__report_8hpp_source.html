108 void parse_assembly_report(std::istream & report)
109 {
110 std::vector<char> report_line;
111 size_t const default_line_buffer_size = 64 * 1024;
112 size_t const assembly_report_column_count = 10;
113 report_line.reserve(default_line_buffer_size);
114 std::vector<std::string> assembly_report_errors; // to contain errors while parsing
115 std::map<std::string,std::set<int>> multiple_occurrence_of_contig_errors;
116
117 int contig_index = 0, line_offset = 1;
118 while (util::readline(report, report_line).size() != 0) {
119 if (report_line[0] == '#') {
120 line_offset++;
121 continue;
122 }
123
124 std::vector<std::string> columns;
125 std::string line{report_line.begin(), report_line.end()};
126 boost::algorithm::trim(line);
127 util::string_split(line, "\t", columns);
128 if (columns.size() != assembly_report_column_count) {
129 std::string error = "Line num " + std::to_string(line_offset + contig_index)
130 + " : Expected " + std::to_string(assembly_report_column_count)
131 + " columns, found " + std::to_string(columns.size()) + "\n";
132 assembly_report_errors.push_back(error);
133 continue;
134 }
135
136 auto contig_synonyms = extract_synonyms(columns);
137 contigs.push_back(contig_synonyms);
138 for (auto contig : contig_synonyms.get_synonyms()) {
139 if (ignore_contig.find(contig) != ignore_contig.end()) {
140 continue;
141 }
142 if (index_map.find(contig) != index_map.end() && index_map[contig] != contig_index) {
143 // if a contig is found in two different lines that would be treated as error
144 multiple_occurrence_of_contig_errors[contig].insert(contig_index + line_offset);
145 multiple_occurrence_of_contig_errors[contig].insert(index_map[contig] + line_offset);
146 continue;
147 }
148 index_map[contig] = contig_index;
149 }
150 contig_index++;
151
152 }
153
154 for (auto contig_error: multiple_occurrence_of_contig_errors) {
155 std::stringstream error;
156 error << "Contig '" + contig_error.first + "' found in multiple lines : ";
157 ebi::util::print_container(error, contig_error.second, "", ",", "");
158 error << std::endl;
159 assembly_report_errors.push_back(error.str());
160 }
161
162 if (!assembly_report_errors.empty()) {
163 std::string error_report = "Some errors occurred while parsing assembly report file \n";
164 for (auto error : assembly_report_errors) {
165 error_report += error;
166 }
167 throw std::runtime_error(error_report);
168 }
169 }
170
171 private:
172 std::map<std::string,int> index_map;
173 std::vector<ContigSynonyms> contigs;
174 /*
175 * set of names that needs to be ignored if present in assembly report
176 */
177 const std::set<std::string> ignore_contig{"na"};
178
179 ContigSynonyms extract_synonyms(std::vector<std::string> & synonyms)
180 {
181 std::vector<int> synonym_indices{0, 4, 6, 9};
182 ContigSynonyms contig_synonyms;
183 for (auto index : synonym_indices) {
184 if (ignore_contig.find(synonyms[index]) != ignore_contig.end()) {
185 continue;
186 }
187 contig_synonyms.add_synonym(synonyms[index]);
188 }
189
190 return contig_synonyms;
191 }

193 };

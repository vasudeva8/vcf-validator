64 std::string handle_test_results(const boost::filesystem::path &path,
65 bool successfully_fixed, const std::stringstream &fixed_vcf)
66 {
67 std::stringstream debug_message;

73 db_path_after += ".debugulator_test." + AFTER_TAG + ".db";
74
75 boost::filesystem::remove(fixed_vcf_path);
76
77 if (!successfully_fixed) {
78 // dump fixed file to manually inspect
79 std::ofstream fixed_file{fixed_vcf_path.string()};
80 fixed_file << fixed_vcf.str();
81 fixed_file.close();
82 debug_message << "This test failed, you can see the related data at: \n" << db_path.string() << "\n";
83 debug_message << db_path_after.string() << "\n" << fixed_vcf_path.string() << std::endl;
84 } else {
85 // clean report dbs
86 boost::filesystem::remove(db_path);
87 boost::filesystem::remove(db_path_after);
88 }
89
90 return debug_message.str();

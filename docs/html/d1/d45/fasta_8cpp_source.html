35{

51 fasta_index = bioio::read_fasta_index(faidx_index);
52 }
53}

122ebi::vcf::fasta::RemoteContig::sequence(const std::string& contig, const size_t pos, const size_t length)
123{
124 if (contigs.count(contig) == 0) {
125 contigs[contig].reset(new ebi::vcf::fasta::ContigFromENA(contig));
126
127 // This contig is not downloaded, try download it from ENA.
128 std::string url = ebi::vcf::ENA_API_FASTA_URL + contig;
129 std::fstream response_stream;
130 response_stream.open(contig+".tmp", std::ios::in | std::ios::out | std::ios::app);
131 ebi::util::open_remote(response_stream, url);
132 response_stream.seekg(0, std::ios::beg);
133
134 std::string line;
135 line.reserve(1024);
136 if (ebi::util::readline(response_stream, line).size() != 0) {
137 if (boost::starts_with(line, ">")) {
138 while (ebi::util::readline(response_stream, line).size() != 0) {
139 ebi::util::remove_end_of_line(line); // ignore linebreak at the end of line
140 contigs[contig]->write(line.c_str(), line.size());
141 }
142 }
143 }
144
145 response_stream.close();
146 boost::filesystem::remove(contig+".tmp");
147 }
148
149 return contigs[contig]->read(pos, length);
150}
